Under the Hood

Orchestration Logic


// backend/src/agents/coordinatorAgent.js

export async function runAgent(runId, repoUrl, ...) {
  // 1. Initialize Swarm
  const repoAnalyzer = new RepoAnalyzerAgent(workspaceDir);
  const testRunner = new TestRunnerAgent();
  const fixGenerator = new FixGeneratorAgent();

  // 2. Clone & Discover
  const repoPath = await repoAnalyzer.cloneRepository(repoUrl);
  const testFiles = await repoAnalyzer.discoverTestFiles(repoPath);

  // 3. Iterative Healing Loop
  while (iteration < retryLimit) {
    const failures = await testRunner.runTests(repoPath);
    
    if (failures.length === 0) return { status: 'PASSED' };

    // 4. Generate & Apply Semantic Fixes
    for (const failure of failures) {
      const fix = await fixGenerator.generateFix(failure);
      await fixGenerator.applyFix(repoPath, fix);
    }
    
    iteration++;
  }
}
Under the Hood

AI Fix Generation


// backend/src/agents/fixGeneratorAgent.js

async generateFix(failure, context) {
  // Construct prompt with stack trace + file context
  const prompt = `
    Error: ${failure.message}
    File: ${context.fileContent}
    Fix this specific error. Return JSON only.
  `;

  // Call LLM (GPT-4)
  const fix = await this.llm.predict(prompt);
  
  // Validate syntax before applying
  if (this.validator.checkSyntax(fix.code)) {
    return fix;
  }
}
